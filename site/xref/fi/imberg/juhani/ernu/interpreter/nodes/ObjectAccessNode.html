3   import fi.imberg.juhani.ernu.interpreter.Environment;
4   import fi.imberg.juhani.ernu.interpreter.exceptions.RuntimeException;
5   import fi.imberg.juhani.ernu.interpreter.interfaces.Identifier;
6   import fi.imberg.juhani.ernu.interpreter.interfaces.Node;
7   import fi.imberg.juhani.ernu.interpreter.interfaces.Object;
8   
9   /**
10   * Accesses an object and either brings results or assigns nodes to the object.
11   */
12  public class ObjectAccessNode implements Node, Identifier {
13      /**
14       * The node that must resolve into an object to be accessed
15       */
16      private final Node object;
17      /**
18       * The thing thats accessed in the object
19       */
20      private final Node thing;
21  
22      /**
23       * @param obj   The node that must resolve into an object to be accessed
24       * @param thing The thing thats accessed in the object
25       */
26      public ObjectAccessNode(Node obj, Node thing) {
27          this.object = obj;
28          this.thing = thing;
29      }
30  
31      @Override
32      public Node getValue(Environment environment) throws RuntimeException {
33          Node val = object.getValue(environment);
34          if (!(val instanceof Object)) {
35              throw new RuntimeException("Node not an object!");
36          }
37          Object objectNode = (Object) val;
38          return objectNode.getAttribute(thing.toString());
39      }
40  
41      @Override
42      public void setValue(Environment environment, Node value) throws RuntimeException {
43          Node val = object.getValue(environment);
44          if (!(val instanceof Object)) {
45              throw new RuntimeException("Node not an object!");
46          }
47          Object objectNode = (Object) val;
48          objectNode.setAttribute(thing.toString(), value.getValue(environment));
49      }
50  
51      @Override
52      public String toString() {
53          return object.toString() + "." + thing;
54      }
55  
56      public Node getObject() {
57          return object;
58      }
59  }
