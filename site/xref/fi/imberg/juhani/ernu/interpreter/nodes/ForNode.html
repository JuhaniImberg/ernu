7   /**
8    * A for node represents a for loop.
9    */
10  public class ForNode implements Node {
11      /**
12       * The node that gets assigned the generators current value each loop
13       */
14      private final Node identifier;
15      /**
16       * Something that resolves into a list for iteration
17       */
18      private final Node generator;
19      /**
20       * The body that gets executed each loop
21       */
22      private final BlockNode body;
23  
24      /**
25       * @param identifier The node that gets assigned the generators current value each loop
26       * @param generator  Something that resolves into a list for iteration
27       * @param body       The body that gets executed each loop
28       */
29      public ForNode(Node identifier, Node generator, BlockNode body) {
30          this.identifier = identifier;
31          this.generator = generator;
32          this.body = body;
33      }
34  
35      @Override
36      public String toString() {
37          return "(for " + identifier + " -> " + generator + " " + body + ")";
38      }
39  
40      @Override
41      public Node getValue(Environment environment) throws RuntimeException {
42          Node generated = generator.getValue(environment);
43          Environment local = environment;
44          if (!(generated instanceof ArrayNode)) {
45              throw new RuntimeException("Generated not a array");
46          }
47          if (!(identifier instanceof IdentifierNode)) {
48              throw new RuntimeException("Not identifier");
49          }
50          ArrayNode generatedArray = (ArrayNode) generated;
51          Node value = new NullNode();
52          for (int i = 0; i < generatedArray.length(); i++) {
53              local.addSymbol(identifier.toString(), generatedArray.get(i));
54              value = body.getValue(local);
55          }
56          return value;
57      }
58  }
