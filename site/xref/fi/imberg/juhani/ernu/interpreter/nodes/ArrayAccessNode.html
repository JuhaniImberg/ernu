10  import java.util.List;
11  
12  /**
13   * Accesses an array and either brings results or assigns nodes to the array.
14   */
15  public class ArrayAccessNode implements Node, Identifier {
16      /**
17       * A node that resolves to a Sequence.
18       */
19      private final Node identifier;
20      /**
21       * The arguments for the access, since array slicing is a real thing.
22       */
23      private final List<Node> args;
24  
25      /**
26       * @param identifier A node that resolves to a Sequence.
27       * @param args       The arguments for the access, since array slicing is a real thing.
28       */
29      public ArrayAccessNode(Node identifier, List<Node> args) {
30          this.identifier = identifier;
31          this.args = args;
32      }
33  
34      @Override
35      public String toString() {
36          return identifier + "[" + args + "]";
37      }
38  
39      private NumberNode getRealIndex(Environment environment) throws RuntimeException {
40          Node realIndex = args.get(0).getValue(environment);
41          if (!(realIndex instanceof NumberNode)) {
42              throw new RuntimeException("Index must be an integer.");
43          }
44          return (NumberNode) realIndex;
45      }
46  
47      private Sequence getSequence(Environment environment) throws RuntimeException {
48          Node value = identifier.getValue(environment);
49          if (!(value instanceof Sequence)) {
50              throw new RuntimeException("Identifier is not a sequence.");
51          }
52          return (Sequence) value;
53      }
54  
55      private int getActuallyRealPos(Sequence sequence, NumberNode numberNode) throws RuntimeException {
56          int pos = numberNode.getInteger();
57          if (pos >= 0) {
58              if (pos >= sequence.length()) {
59                  throw new RuntimeException("Array access out of bounds.");
60              }
61              return pos;
62          } else {
63              if (Math.abs(pos) - 1 >= sequence.length()) {
64                  throw new RuntimeException("Array access out of bounds.");
65              }
66              return sequence.length() - Math.abs(pos);
67          }
68      }
69  
70      private int getIntOfArg(int i, int def, Environment environment) throws RuntimeException {
71          if (args.size() <= i) {
72              return def;
73          }
74          Node node = args.get(i).getValue(environment);
75          if (node instanceof NullNode) {
76              return def;
77          }
78          if (!(node instanceof NumberNode)) {
79              throw new RuntimeException("Index must be an integer.");
80          }
81          return ((NumberNode) node).getInteger();
82      }
83  
84      @Override
85      public Node getValue(Environment environment) throws RuntimeException {
86          Sequence sequence = getSequence(environment);
87          if (args.size() == 1) {
88              NumberNode realIndexInteger = getRealIndex(environment);
89              return sequence.get(getActuallyRealPos(sequence, realIndexInteger));
90          } else {
91              int start = getIntOfArg(0, 0, environment);
92              int end = getIntOfArg(1, sequence.length(), environment);
93              int step = getIntOfArg(2, 1, environment);
94              if (start < 0) {
95                  start = sequence.length() + start;
96              }
97              if (end < 0) {
98                  end = sequence.length() + end;
99              }
100             Appendable newSequence = (Appendable) sequence.newEmpty();
101             if (step >= 0) {
102                 for (int i = start; i < end; i += step) {
103                     newSequence = (Appendable) newSequence.append(sequence.get(i));
104                 }
105             } else {
106                 for (int i = end - 1; i >= start; i += step) {
107                     newSequence = (Appendable) newSequence.append(sequence.get(i));
108                 }
109             }
110             return (Node) newSequence;
111         }
112     }
113 
114     @Override
115     public void setValue(Environment environment, Node value) throws RuntimeException {
116         if (args.size() > 1) {
117             throw new RuntimeException("slice assign is not supported");
118         }
119         NumberNode realIndexInteger = getRealIndex(environment);
120         Sequence sequence = getSequence(environment);
121         sequence.set(getActuallyRealPos(sequence, realIndexInteger), value);
122     }
123 }
