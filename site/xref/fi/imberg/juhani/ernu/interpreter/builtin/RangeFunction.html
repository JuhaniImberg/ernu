12  /**
13   * Returns a list of numbers in an increasing order.
14   */
15  public class RangeFunction extends BuiltinFunction {
16      public RangeFunction() {
17          super("Creates a range.");
18      }
19  
20      @Override
21      public Node call(Environment environment, List<Node> arguments) throws RuntimeException {
22          if (arguments.size() == 0) {
23              throw new RuntimeException("range takes at least one argument");
24          }
25          if (arguments.size() > 3) {
26              throw new RuntimeException("range takes three arguments at most");
27          }
28          List<NumberNode> args = new ArrayList<>();
29          for (Node node : arguments) {
30              Node v = node.getValue(environment);
31              if (!(v instanceof NumberNode)) {
32                  throw new RuntimeException("range takes number nodes as arguments");
33              }
34              args.add((NumberNode) v);
35          }
36  
37          List<Node> values = new ArrayList<>();
38          int to = args.get(0).getInteger();
39          int from = 0;
40          int step = 1;
41          if (args.size() >= 2) {
42              from = args.get(1).getInteger();
43              int asd = to;
44              to = from;
45              from = asd;
46          }
47          if (args.size() == 3) {
48              step = args.get(2).getInteger();
49          }
50          if (step > 0) {
51              for (int i = from; i < to; i += step) {
52                  values.add(new NumberNode(i));
53              }
54          } else {
55              for (int i = to; i > from; i += step) {
56                  values.add(new NumberNode(i));
57              }
58          }
59          return new ArrayNode(values);
60      }
61  }
