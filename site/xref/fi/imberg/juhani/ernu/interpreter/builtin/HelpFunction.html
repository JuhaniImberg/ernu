10  /**
11   * Returns the __doc__ attribute of an object.
12   */
13  public class HelpFunction extends BuiltinFunction {
14      public HelpFunction() {
15          super("Helps with something.");
16      }
17  
18      @Override
19      public Node call(Environment environment, List<Node> arguments) throws RuntimeException {
20          if (arguments.size() != 1) {
21              throw new RuntimeException("help takes one argument only");
22          }
23          Node node = arguments.get(0).getValue(environment);
24  
25          if (!(node instanceof Object)) {
26              throw new RuntimeException("helps argument must be an object");
27          }
28          return ((Object) node).getAttribute("__doc__");
29      }
30  }
