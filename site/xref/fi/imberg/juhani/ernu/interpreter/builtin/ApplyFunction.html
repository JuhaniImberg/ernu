12  /**
13   * Applies a args list to a function.
14   */
15  public class ApplyFunction extends BuiltinFunction {
16      public ApplyFunction() {
17          super("Applies a list of arguments to a function");
18      }
19  
20      @Override
21      public Node call(Environment environment, List<Node> arguments) throws RuntimeException {
22          if (arguments.size() != 2) {
23              throw new RuntimeException("apply takes two arguments");
24          }
25          Node node = arguments.get(0).getValue(environment);
26          if (!(node instanceof FunctionNode)) {
27              throw new RuntimeException("applys first argument must be a function");
28          }
29          Node args = arguments.get(1).getValue(environment);
30          if (!(args instanceof ArrayNode)) {
31              throw new RuntimeException("applys second argument must be an array");
32          }
33          return (new CallNode(node, ((ArrayNode) args).getValues())).getValue(environment);
34      }
35  }
