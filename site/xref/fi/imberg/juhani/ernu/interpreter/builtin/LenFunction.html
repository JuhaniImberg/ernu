11  /**
12   * Returns the length of a sequence.
13   */
14  public class LenFunction extends BuiltinFunction {
15      public LenFunction() {
16          super("Returns the length of something.");
17      }
18  
19      @Override
20      public Node call(Environment environment, List<Node> arguments) throws RuntimeException {
21          if (arguments.size() != 1) {
22              throw new RuntimeException("len takes exactly one argument");
23          }
24          Node arg = arguments.get(0).getValue(environment);
25  
26          if (!(arg instanceof Sequence)) {
27              throw new RuntimeException("lens argument should be a sequence");
28          }
29          return new NumberNode(((Sequence) arg).length());
30      }
31  }
