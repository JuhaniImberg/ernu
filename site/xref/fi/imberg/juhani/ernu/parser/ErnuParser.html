3   import fi.imberg.juhani.ernu.parser.parsers.*;

5   /**
6    * The actual parser, with all of the sub parsers added.
7    */
8   public class ErnuParser extends Parser {
9       public ErnuParser(Tokenizer tokenizer) {
10          super(tokenizer);
11  
12          addPrefixParser(TokenType.IDENTIFIER, new IdentifierParser());
13          addPrefixParser(TokenType.NUMBER, new NumberParser());
14          addPrefixParser(TokenType.LPAREN, new GroupingParser());
15          addPrefixParser(TokenType.FUNCTION, new FunctionParser());
16          addPrefixParser(TokenType.STRING, new StringParser());
17          addPrefixParser(TokenType.IF, new IfParser());
18          addPrefixParser(TokenType.WHILE, new WhileParser());
19          addPrefixParser(TokenType.FOR, new ForParser());
20          addPrefixParser(TokenType.MATCH, new MatchParser());
21          addPrefixParser(TokenType.CLASS, new ClassParser());
22          addPrefixParser(TokenType.RETURN, new ReturnParser());
23          addPrefixParser(TokenType.LBRACKET, new ArrayParser());
24          addPrefixParser(TokenType.TRUE, new BooleanParser());
25          addPrefixParser(TokenType.FALSE, new BooleanParser());
26          addPrefixParser(TokenType.IMPORT, new ImportParser());
27          addPrefixParser(TokenType.FROM, new ImportParser());
28  
29          addPrefixParser(TokenType.NOT, new PrefixOperatorParser());
30          addPrefixParser(TokenType.SUB, new PrefixOperatorParser());
31  
32          addInfixParser(TokenType.LPAREN, new CallParser());
33          addInfixParser(TokenType.LBRACKET, new ArrayAccessParser());
34          addInfixParser(TokenType.OBJECT_ACCESS, new ObjectAccessParser());
35  
36          addInfixParser(TokenType.SET, new AssignmentParser());
37          addInfixParser(TokenType.ADDSET, new AssignmentParser());
38          addInfixParser(TokenType.SUBSET, new AssignmentParser());
39          addInfixParser(TokenType.MULSET, new AssignmentParser());
40          addInfixParser(TokenType.DIVSET, new AssignmentParser());
41          addInfixParser(TokenType.MODSET, new AssignmentParser());
42  
43          addInfixParser(TokenType.OR, new OperatorParser(Precedence.OR, false));
44          addInfixParser(TokenType.AND, new OperatorParser(Precedence.AND, false));
45  
46          addInfixParser(TokenType.EQ, new OperatorParser(Precedence.EQUALITY, false));
47          addInfixParser(TokenType.NOTEQ, new OperatorParser(Precedence.EQUALITY, false));
48  
49          addInfixParser(TokenType.GT, new OperatorParser(Precedence.COMPARISON, false));
50          addInfixParser(TokenType.LT, new OperatorParser(Precedence.COMPARISON, false));
51          addInfixParser(TokenType.GTOE, new OperatorParser(Precedence.COMPARISON, false));
52          addInfixParser(TokenType.LTOE, new OperatorParser(Precedence.COMPARISON, false));
53  
54          addInfixParser(TokenType.ADD, new OperatorParser(Precedence.SUM, false));
55          addInfixParser(TokenType.SUB, new OperatorParser(Precedence.SUM, false));
56          addInfixParser(TokenType.DIV, new OperatorParser(Precedence.PRODUCT, false));
57          addInfixParser(TokenType.MUL, new OperatorParser(Precedence.PRODUCT, false));
58          addInfixParser(TokenType.MOD, new OperatorParser(Precedence.PRODUCT, false));
59      }
60  }
