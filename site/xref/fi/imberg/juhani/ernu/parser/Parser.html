3   import fi.imberg.juhani.ernu.interpreter.interfaces.Node;
4   import fi.imberg.juhani.ernu.parser.exceptions.LangException;
5   import fi.imberg.juhani.ernu.parser.exceptions.ParsingException;
6   import fi.imberg.juhani.ernu.parser.exceptions.UnexpectedTokenException;
7   import fi.imberg.juhani.ernu.parser.parsers.InfixParser;
8   import fi.imberg.juhani.ernu.parser.parsers.PrefixParser;
9   
10  import java.util.ArrayList;
11  import java.util.HashMap;
12  
13  /**
14   * Takes a bunch of tokens and builds a tree out of them.
15   */
16  public class Parser {
17      private final Tokenizer tokenizer;
18      private final ArrayList<Token> buffer;
19      private final HashMap<TokenType, PrefixParser> prefixParsers;
20      private final HashMap<TokenType, InfixParser> infixParsers;
21  
22      /**
23       * @param tokenizer A tokenizer that has already parsed some text.
24       */
25      public Parser(Tokenizer tokenizer) {
26          this.tokenizer = tokenizer;
27          this.buffer = new ArrayList<>();
28          this.prefixParsers = new HashMap<>();
29          this.infixParsers = new HashMap<>();
30      }
31  
32      public void addInfixParser(TokenType tokenType, InfixParser parser) {
33          this.infixParsers.put(tokenType, parser);
34      }
35  
36      public void addPrefixParser(TokenType tokenType, PrefixParser parser) {
37          this.prefixParsers.put(tokenType, parser);
38      }
39  
40      /**
41       * @param precedence Limits when to stop parsing, if a parser with higher precedence
42       *                   is encountered, the parsing is stopped.
43       * @return A node which probably has some children, a small tree basically.
44       * @throws LangException If an unexpected token is encountered
45       */
46      public Node parseNode(int precedence) throws LangException {
47          Token current = consume();
48          if (current.getType() == TokenType.EOL) {
49              return null;
50          }
51          PrefixParser prefixParser = prefixParsers.get(current.getType());
52          if (prefixParser == null) {
53              throw new ParsingException("Unexpected token.", current);
54          }
55          Node left = prefixParser.parse(this, current);
56          while (precedence < getPrecedence()) {
57              current = consume();
58              InfixParser infixParser = infixParsers.get(current.getType());
59              left = infixParser.parse(this, left, current);
60          }
61          return left;
62      }
63  
64      /**
65       * parseNode with the precedence of 0
66       *
67       * @return A node which probably has some children, a small tree basically.
68       * @throws LangException If an unexpected token is encountered
69       */
70      public Node parseNode() throws LangException {
71          return parseNode(0);
72      }
73  
74      /**
75       * Consumes the next token if it's of the type
76       *
77       * @param type Token type to match against
78       * @return True if the next token was of type
79       */
80      public boolean match(TokenType type) {
81          Token token = lookAhead(0);
82          if (token.getType() != type) {
83              return false;
84          }
85          consume();
86          return true;
87      }
88  
89      /**
90       * @param type Token type to match against
91       * @return True if the next token was of type
92       */
93      public boolean isNext(TokenType type) {
94          Token token = lookAhead(0);
95          return token.getType() == type;
96      }
97  
98      /**
99       * @param type Token type to match against
100      * @return A token that is of the type
101      * @throws UnexpectedTokenException If the next token is not of the type
102      */
103     public Token consume(TokenType type) throws UnexpectedTokenException {
104         Token token = lookAhead(0);
105         if (token.getType() != type) {
106             throw new UnexpectedTokenException(type, token);
107         }
108         return consume();
109     }
110 
111     /**
112      * @return The next token
113      */
114     public Token consume() {
115         lookAhead(0);
116         return buffer.remove(0);
117     }
118 
119     /**
120      * @return The precedence of the next tokens parser type
121      */
122     public int getPrecedence() {
123         InfixParser parser = infixParsers.get(lookAhead(0).getType());
124         if (parser != null) {
125             return parser.getPrecedence();
126         }
127         return 0;
128     }
129 
130     /**
131      * @param number How many tokens are buffered from the tokenizer
132      * @return The nth next token
133      */
134     private Token lookAhead(int number) {
135         while (number >= buffer.size()) {
136             buffer.add(tokenizer.nextToken());
137         }
138         return buffer.get(number);
139     }
140 }
