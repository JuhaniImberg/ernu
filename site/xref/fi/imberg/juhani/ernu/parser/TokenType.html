3   /**
4    * Enum of all of the possible token types.
5    */
6   public enum TokenType {
7       LPAREN("\\("),
8       RPAREN("\\)"),
9       LBRACKET("\\["),
10      RBRACKET("\\]"),
11      NUMBER("\\-?[0-9]+\\.?[0-9]*"),
12      FUNCTION("function"),
13      IF("if"),
14      DO("do"),
15      ELSE("else"),
16      END("end"),
17      RETURN("return"),
18      ELSEIF("elseif"),
19      MATCH("match"),
20      WHILE("while"),
21      BREAK("break"),
22      CASE("case"),
23      FOR("for"),
24      IN("in"),
25      CLASS("class"),
26      IMPORT("import"),
27      FROM("from"),
28      AS("as"),
29      TRUE("true"),
30      FALSE("false"),
31      AND("and|\\&\\&"),
32      OR("or|\\|\\|"),
33      NOT("not|\\!"),
34      IDENTIFIER("[A-Za-z_]+[A-Za-z0-9_\\.]*[?\\']{0,1}"),
35      STRING("\".+\""),
36      EOL("\n|\\;"),
37      EQ("=="),
38      SET("="),
39      GT("\\>"),
40      LT("\\<"),
41      GTOE("\\>="),
42      LTOE("\\<="),
43      NOTEQ("\\!="),

46      SUB("\\-"),
47      DIV("\\/"),
48      MUL("\\*"),
49      MODSET("%="),
50      ADDSET("\\+="),
51      SUBSET("\\-="),
52      DIVSET("\\/="),
53      MULSET("\\*="),
54      COMMA(","),
55      OBJECT_ACCESS("\\."),
56      SLICE_SEPARATOR("\\:"),
57      NONE(""),
58      EOF("");
59  
60      /**
61       * A regex that a matching tokens content should match
62       */
63      private String match;
64  
65      TokenType(String match) {
66          this.match = match;
67      }
68  
69      /**
70       * Finds the first suitable type for the token, and marks the token as being of the found type.
71       *
72       * @param token The token to be matched against.
73       */
74      public static void matchType(Token token) {
75          if (token.getType() != TokenType.NONE) {
76              return;
77          }
78          String content = token.getContent();
79          for (TokenType ttype : values()) {
80              if (content.matches(ttype.getMatch())) {
81                  token.setType(ttype);
82                  return;
83              }
84          }
85      }
86  
87      public String getMatch() {
88          return match;
89      }
90  }
