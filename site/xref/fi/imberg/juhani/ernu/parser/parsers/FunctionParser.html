15  /**
16   * Parses an expression for a function. For example function (i) return i end
17   */
18  public class FunctionParser implements PrefixParser {
19      @Override
20      public Node parse(Parser parser, Token token) throws LangException {
21          List<Node> args = new ArrayList<>();
22          List<Node> nodes = new ArrayList<>();
23          String doc = "";
24  
25          parser.consume(TokenType.LPAREN);
26          if (!parser.match(TokenType.RPAREN)) {
27              do {
28                  args.add(parser.parseNode());
29              } while (parser.match(TokenType.COMMA));
30              parser.consume(TokenType.RPAREN);
31          }
32  
33          parser.match(TokenType.EOL);
34  
35          if (parser.isNext(TokenType.STRING)) {
36              doc = ((StringNode) parser.parseNode()).getStringLiteral();
37              parser.match(TokenType.EOL);
38          }
39  
40          if (!parser.match(TokenType.END)) {
41              do {
42                  nodes.add(parser.parseNode());
43                  parser.match(TokenType.EOL);
44              } while (!parser.match(TokenType.END));
45          }
46  
47          return new FunctionNode(args, doc, new BlockNode(nodes));
48      }
49  }
