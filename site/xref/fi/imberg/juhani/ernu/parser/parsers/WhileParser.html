14  /**
15   * Parses an expression for a while statement. For example while true do print("asd") end
16   */
17  public class WhileParser implements PrefixParser {
18      @Override
19      public Node parse(Parser parser, Token token) throws LangException {
20          Node condition = parser.parseNode();
21          List<Node> body = new ArrayList<>();
22          parser.consume(TokenType.DO);
23          parser.match(TokenType.EOL);
24          if (!parser.match(TokenType.END)) {
25              do {
26                  body.add(parser.parseNode());
27                  parser.match(TokenType.EOL);
28              } while (!parser.match(TokenType.END));
29          }
30          return new WhileNode(condition, new BlockNode(body));
31      }
32  }
