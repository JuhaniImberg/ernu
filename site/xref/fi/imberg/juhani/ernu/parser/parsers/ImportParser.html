13  /**
14   * Parses an expression for an import statement. For example from math import sqrt
15   */
16  public class ImportParser implements PrefixParser {
17      @Override
18      public Node parse(Parser parser, Token token) throws LangException {
19          String what = "";
20          String as = "";
21          List<String> subs = new ArrayList<>();
22          while (!(parser.isNext(TokenType.EOL) || parser.isNext(TokenType.AS) || parser.isNext(TokenType.IMPORT))) {
23              what += parser.consume().getContent();
24          }
25          if (parser.match(TokenType.AS)) {
26              as = parser.consume().getContent();
27          } else if (parser.match(TokenType.IMPORT)) {
28              do {
29                  subs.add(parser.consume(TokenType.IDENTIFIER).getContent());
30              } while (parser.match(TokenType.COMMA));
31          }
32          return new ImportNode(what, as, subs);
33      }
34  }
