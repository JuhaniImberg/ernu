16  public class ClassNodeTest {
17      @Rule
18      public ExpectedException thrown = ExpectedException.none();
19      Environment environment;
20  
21      @Before
22      public void init() {
23          environment = new Environment(false, null, "test");
24      }
25  
26      @After
27      public void destroy() {
28          environment = null;
29      }
30  
31      public ClassNode createClassNode(String source) throws ErnuException {
32          Tokenizer tokenizer = new Tokenizer();
33          ErnuParser parser = new ErnuParser(tokenizer);
34          tokenizer.tokenize(source + "\n\n");
35          Node node = parser.parseNode();
36          Assert.assertTrue(node instanceof ClassNode);
37          node.getValue(environment);
38          return (ClassNode) node;
39      }
40  
41      public CallNode createCallNode(Node node, Node[] args) {
42          List<Node> argsList = new ArrayList<>();
43          Collections.addAll(argsList, args);
44          return new CallNode(node, argsList);
45      }
46  
47      public ObjectAccessNode createAccessNode(Node node, String thing) {
48          return new ObjectAccessNode(node, new IdentifierNode(thing));
49      }
50  
51      public void doSimpleAttributeTest(Node node, String thing, Node value) throws RuntimeException {
52          ObjectAccessNode objectAccessNode = createAccessNode(node, thing);
53          Node ret = objectAccessNode.getValue(environment);
54          Assert.assertEquals(value, ret);
55      }
56  
57      @Test
58      public void simpleAttribute() throws ErnuException {
59          ClassNode classNode = createClassNode("class \"test test test\"; v = 3 end");
60          doSimpleAttributeTest(classNode, "v", new NumberNode(3));
61          doSimpleAttributeTest(classNode, "__doc__", new StringNode("test test test"));
62      }
63  
64      @Test
65      public void simpleInitTest() throws ErnuException {
66          ClassNode classNode = createClassNode("class v = 3 end");
67          Node node = classNode.call(environment, new ArrayList<>());
68          Assert.assertTrue(node instanceof ObjectNode);
69          doSimpleAttributeTest(node, "v", new NumberNode(3));
70      }
71  
72      @Test
73      public void advancedInitTest() throws ErnuException {
74          ClassNode classNode = createClassNode("class init = function (this, val) this.val = val end end");
75          List<Node> args = new ArrayList<>();
76          args.add(new NumberNode(4));
77          Node node = classNode.call(environment, args);
78          Assert.assertTrue(node instanceof ObjectNode);
79          doSimpleAttributeTest(node, "val", new NumberNode(4));
80      }
81  
82      @Test
83      public void methodTest() throws ErnuException {
84          ClassNode classNode = createClassNode("class val = 0; inc = function (this) this.val += 1 end end");
85          Node node = classNode.call(environment, new ArrayList<>());
86          Assert.assertTrue(node instanceof ObjectNode);
87          doSimpleAttributeTest(node, "val", new NumberNode(0));
88          CallNode callNode = new CallNode(createAccessNode(node, "inc"), new ArrayList<>());
89          callNode.getValue(environment);
90          doSimpleAttributeTest(node, "val", new NumberNode(1));
91      }
