15  public class NumFunctionTest {
16      @Rule
17      public ExpectedException thrown = ExpectedException.none();
18      Environment environment;
19      NumFunction numFunction;
20  
21      @Before
22      public void init() {
23          environment = new Environment(false, null, "test");
24          numFunction = new NumFunction();
25      }
26  
27      @After
28      public void destroy() {
29          environment = null;
30          numFunction = null;
31      }
32  
33      @Test
34      public void needsOneArgument() throws RuntimeException {
35          thrown.expect(fi.imberg.juhani.ernu.interpreter.exceptions.RuntimeException.class);
36          thrown.expectMessage("num takes one argument only");
37  
38          numFunction.call(environment, new ArrayList<>());
39      }
40  
41      @Test
42      public void argumentNeedsToBeString() throws RuntimeException {
43          thrown.expect(RuntimeException.class);
44          thrown.expectMessage("num takes a string only");
45  
46          List<Node> nodeList = new ArrayList<>();
47          nodeList.add(new NullNode());
48  
49          numFunction.call(environment, nodeList);
50      }
51  
52      @Test
53      public void works() throws RuntimeException {
54          List<Node> nodeList = new ArrayList<>();
55          nodeList.add(new StringNode("3"));
56  
57          Node node = numFunction.call(environment, nodeList);
58          Assert.assertTrue(node instanceof NumberNode);
59          Assert.assertEquals(3, ((NumberNode) node).getInteger());
60      }
